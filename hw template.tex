\begin{algorithm}[H]
    \caption{求和算法}
    \begin{algorithmic}[1]
        \Require 一个整数列表 $L$
        \Ensure 列表中所有元素的和
        \State total $\gets 0$
        \For{每个元素 $e$ 在列表 $L$ 中}
            \State total $\gets$ total $+ e$
        \EndFor
        \State \Return total
    \end{algorithmic}
\end{algorithm}
